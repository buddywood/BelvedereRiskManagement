```typescript
export type RiskSeverity = 'critical' | 'moderate' | 'low';

export interface RiskIndicator {
  familyId: string;
  familyName: string;
  categorySlug: string;
  categoryName: string;
  score: number;
  severity: RiskSeverity;
  weight: number; // pillar weight for context
  assessmentId: string;
  assessmentDate: string;
}

export interface FamilyRiskSummary {
  familyId: string;
  familyName: string;
  overallScore: number;
  topRisks: RiskIndicator[]; // top 3 lowest-scoring pillars
  assessmentCount: number;
  latestAssessmentDate: string;
}

export interface PortfolioIntelligence {
  totalFamilies: number;
  familiesAtRisk: number; // families with at least one critical or moderate risk
  criticalCount: number; // total critical risks across portfolio
  familyRiskSummaries: FamilyRiskSummary[];
  portfolioRisks: RiskIndicator[]; // all risks across portfolio, sorted by severity then score
  risksByCategory: Record<string, number>; // count of risks per governance category
}
```

Create `src/lib/intelligence/queries.ts` with `"server-only"` guard:

1. Import PILLAR_WEIGHTS and CATEGORY_LABELS from analytics formatters. Note: PILLAR_WEIGHTS is currently defined in `src/lib/analytics/queries.ts` as a const - you'll need to either export it from there or duplicate the weights object. Prefer exporting from queries.ts if the export doesn't break server-only constraints. If it does, duplicate the weights.

2. `getSeverity(score: number): RiskSeverity` - Returns 'critical' if score <= 3.0, 'moderate' if score <= 5.0, 'low' otherwise.

3. `getTopRisksForFamily(clientId: string, advisorProfileId: string): Promise<FamilyRiskSummary>` - Verifies advisor-client relationship via `clientAdvisorAssignment` (ACTIVE status). Gets latest completed assessment with pillar scores. Calculates weighted overall score using PILLAR_WEIGHTS. Sorts pillars by score ascending, takes bottom 3 as top risks. Returns FamilyRiskSummary with risk indicators including severity, category name, weight.

4. `getPortfolioIntelligence(advisorProfileId: string): Promise<PortfolioIntelligence>` - Gets ALL active client assignments for advisor. For each client with completed assessments, calls internal risk calculation (not the exported function - inline the logic to avoid N+1 advisor verification queries). Aggregates into PortfolioIntelligence: counts families at risk, counts critical issues, builds sorted portfolio risk list (critical first, then by score ascending), builds risksByCategory distribution map.

Use the same Prisma patterns as `src/lib/dashboard/queries.ts` and `src/lib/analytics/queries.ts` for advisor scoping and assessment retrieval.
