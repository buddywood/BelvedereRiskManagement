```typescript
export type RiskSeverity = 'critical' | 'moderate' | 'low';

export interface RiskIndicator {
  familyId: string;
  familyName: string;
  categorySlug: string;
  categoryName: string;
  score: number;
  severity: RiskSeverity;
  weight: number; // pillar weight for context
  assessmentId: string;
  assessmentDate: string;
}

export interface FamilyRiskSummary {
  familyId: string;
  familyName: string;
  overallScore: number;
  topRisks: RiskIndicator[]; // top 3 lowest-scoring pillars
  assessmentCount: number;
  latestAssessmentDate: string;
}

export interface PortfolioIntelligence {
  totalFamilies: number;
  familiesAtRisk: number; // families with at least one critical or moderate risk
  criticalCount: number; // total critical risks across portfolio
  familyRiskSummaries: FamilyRiskSummary[];
  portfolioRisks: RiskIndicator[]; // all risks across portfolio, sorted by severity then score
  risksByCategory: Record<string, number>; // count of risks per governance category
}
```

Create `src/lib/intelligence/queries.ts` with `"server-only"` guard:

1. Import PILLAR_WEIGHTS and CATEGORY_LABELS from analytics formatters. Note: PILLAR_WEIGHTS is currently defined in `src/lib/analytics/queries.ts` as a const - you'll need to either export it from there or duplicate the weights object. Prefer exporting from queries.ts if the export doesn't break server-only constraints. If it does, duplicate the weights.

2. `getSeverity(score: number): RiskSeverity` - Returns 'critical' if score <= 3.0, 'moderate' if score <= 5.0, 'low' otherwise.

3. `getTopRisksForFamily(clientId: string, advisorProfileId: string): Promise<FamilyRiskSummary>` - Core risk identification algorithm (INTEL-01):

   **Step-by-step algorithm:**
   a. Verify advisor-client relationship via `clientAdvisorAssignment` where `clientId` and `advisorId` match with ACTIVE status. Return null if no relationship.
   b. Query latest completed Assessment for this user (`prisma.assessment.findFirst` where `userId = clientId`, `status = 'completed'`, ordered by `completedAt desc`), including its `scores` relation (PillarScore[]).
   c. For each PillarScore in `assessment.scores`, create a RiskIndicator: look up `PILLAR_WEIGHTS[pillarScore.pillar]` for weight, `CATEGORY_LABELS[pillarScore.pillar]` for display name, call `getSeverity(pillarScore.score)` for severity.
   d. **Sort all RiskIndicators by score ASCENDING** (lowest scores first = highest risk).
   e. **Slice the first 3 elements** - these are the top 3 risks (INTEL-01).
   f. Calculate weighted overall score using `calculateWeightedScore()` (same logic as analytics/queries.ts).
   g. Return FamilyRiskSummary with `topRisks` containing exactly the 3 lowest-scoring pillars.

4. `getPortfolioIntelligence(advisorProfileId: string): Promise<PortfolioIntelligence>` - Gets ALL active client assignments for advisor. For each client with completed assessments, runs the same risk calculation inline (not calling getTopRisksForFamily to avoid N+1 advisor verification queries). Aggregates into PortfolioIntelligence: counts families at risk (any pillar with severity critical or moderate), counts total critical issues, builds sorted portfolio risk list (sorted by severity order [critical=0, moderate=1, low=2] then by score ascending within same severity), builds risksByCategory distribution map counting how many families have critical or moderate risk in each pillar.

Use the same Prisma patterns as `src/lib/dashboard/queries.ts` and `src/lib/analytics/queries.ts` for advisor scoping and assessment retrieval. Note: the schema uses `clientId` (User.id) not `familyId` in ClientAdvisorAssignment.
